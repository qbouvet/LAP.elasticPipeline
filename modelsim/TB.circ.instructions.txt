# adrA adrB adrC opx   imm5 op

#
# immediate type instruction
# adrA adrB imm16           op 
# |---||---||--------------||----|
# 
# break command
# 00000000000000000000000000111111
#
# load values 1 - 5 in the first 5 registers :
|---||---||--------------||----|
00001000000000000000000001100000
00010000000000000000000010100000
00011000000000000000000011100000
00100000000000000000000100100000
00101000000000000000000101100000

|---||---||--------------||----|
